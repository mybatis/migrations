82    private void assertStatus(ClassLoader classLoader) {
83      ConnectionProvider provider = new JdbcConnectionProvider(classLoader, driver, url, username, password);
84      StatusOperation status = new StatusOperation().operate(provider, migrationsLoader, dbOption, new PrintStream(out));
85      assertEquals(0, status.getAppliedCount());
86      assertEquals(3, status.getPendingCount());
87      assertEquals(3, status.getCurrentStatus().size());
88    }

147   @Test
148   void testUpWithHook() throws Exception {
149     final PrintStream printStream = new PrintStream(out);
150     MigrationHook hook = new MigrationHook() {
151       @Override
152       public void beforeEach(Map<String, Object> bindingMap) {
153         printStream.println("<BEFORE_EACH>");
154       }
155 
156       @Override
157       public void before(Map<String, Object> bindingMap) {
158         printStream.println("<BEFORE>");
159       }
160 
161       @Override
162       public void afterEach(Map<String, Object> bindingMap) {
163         printStream.println("<AFTER_EACH>");
164       }
165 
166       @Override
167       public void after(Map<String, Object> bindingMap) {
168         printStream.println("<AFTER>");
169       }
170     };
171     new UpOperation(3).operate(connectionProvider, migrationsLoader, dbOption, printStream, hook);
172     String output = out.toString("utf-8");
173     assertEquals(1, TestUtil.countStr(output, "<BEFORE>"));
174     assertEquals(3, TestUtil.countStr(output, "<BEFORE_EACH>"));
175     assertEquals(3, TestUtil.countStr(output, "<AFTER_EACH>"));
176     assertEquals(1, TestUtil.countStr(output, "<AFTER>"));
177     out.reset();
178     new DownOperation(2).operate(connectionProvider, migrationsLoader, dbOption, printStream, hook);
179     output = out.toString("utf-8");
180     assertEquals(1, TestUtil.countStr(output, "<BEFORE>"));
181     assertEquals(2, TestUtil.countStr(output, "<BEFORE_EACH>"));
182     assertEquals(2, TestUtil.countStr(output, "<AFTER_EACH>"));
183     assertEquals(1, TestUtil.countStr(output, "<AFTER>"));
184   }

241   protected void assertTableDoesNotExist(ConnectionProvider connectionProvider, String table) {
242     try {
243       runQuery(connectionProvider, "select count(*) from " + table);
244       fail();
245     } catch (SQLException e) {
246       // expected
247     }
248   }

250   protected FileMigrationLoader createMigrationsLoader(String resource) {
251     URL url = getClass().getClassLoader().getResource(resource);
252     File scriptsDir = new File(url.getFile());
253     Properties properties = new Properties();
254     properties.setProperty("changelog", "CHANGELOG");
255     return new FileMigrationLoader(scriptsDir, "utf-8", properties);
256   }
257 
258   protected void runSql(ConnectionProvider provider, String sql) throws SQLException {
259     try (Connection connection = provider.getConnection(); Statement statement = connection.createStatement()) {
260       statement.execute(sql);
261     }
262   }
263 
264   protected String runQuery(ConnectionProvider provider, String query) throws SQLException {
265     try (Connection connection = provider.getConnection(); Statement statement = connection.createStatement();
266         ResultSet rs = statement.executeQuery(query)) {
267       String result = null;
268       if (rs.next()) {
269         result = rs.getString(1);
270       }
271       return result;
272     }
273   }
