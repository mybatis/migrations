363   public static UnpooledDataSource createUnpooledDataSource(String resource) throws IOException {
364     Properties props = Resources.getResourceAsProperties(resource);
365     UnpooledDataSource ds = new UnpooledDataSource();
366     ds.setDriver(props.getProperty("driver"));
367     ds.setUrl(props.getProperty("url"));
368     ds.setUsername(props.getProperty("username"));
369     ds.setPassword(props.getProperty("password"));
370     return ds;
