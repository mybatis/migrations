16  package org.apache.ibatis.migration.system_property;
17  
18  import static org.junit.jupiter.api.Assertions.*;
19  
20  import com.github.stefanbirkner.systemlambda.SystemLambda;
21  
22  import java.io.File;
23  
24  import org.apache.ibatis.io.Resources;
25  import org.apache.ibatis.migration.Migrator;
26  import org.apache.ibatis.migration.utils.TestUtil;
27  import org.junit.jupiter.api.BeforeAll;
28  import org.junit.jupiter.api.Test;
29  
30  class SystemPropertyTest {
31  
32    private static File dir;
33  
34    @BeforeAll
35    public static void init() throws Exception {
36      dir = Resources.getResourceAsFile("org/apache/ibatis/migration/system_property/testdir");
37    }
38  
39    @Test
40    void testSystemProperties() throws Exception {
41      // This test requires two environment variables
42      // MIGRATIONS_VAR3 = bogus_var3
43      // MIGRATIONS_ENVVAR1 = Environment variable 1
44      SystemLambda.restoreSystemProperties(() -> {
45        System.setProperty("MIGRATIONS_DRIVER", "org.hsqldb.jdbcDriver");
46        System.setProperty("username", "Pocahontas");
47        System.setProperty("var1", "Variable 1");
48        System.setProperty("MIGRATIONS_VAR3", "Variable 3");
49        System.setProperty("migrations_var4", "Variable 4");
50        System.setProperty("MIGRATIONS_VAR5", "Variable 5");
51        String output = SystemLambda.tapSystemOut(() -> {
52          Migrator.main(TestUtil.args("--path=" + dir.getAbsolutePath(), "up", "1", "--trace"));
53        });
54        assertTrue(output.contains("SUCCESS"));
55        assertTrue(output.contains("username: Pocahontas"));
56        assertTrue(output.contains("var1: Variable 1"));
57        assertTrue(output.contains("var2: ${var2}"));
58        assertTrue(output.contains("var3: Variable 3"), "System property should overwrite env var");
59        assertTrue(output.contains("var4: Variable 4"));
60        assertTrue(output.contains("var5: Variable 5"));
61        assertTrue(output.contains("Var5: Var5 in properties file"));
62        assertTrue(output.contains("envvar1: Environment variable 1"));
63  
64        Migrator.main(TestUtil.args("--path=" + dir.getAbsolutePath(), "down", "1"));
65      });
66    }
67  }
