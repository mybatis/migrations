31  import org.apache.ibatis.migration.Migrator;
32  import org.apache.ibatis.migration.io.Resources;
33  import org.apache.ibatis.migration.utils.TestUtil;
34  import org.apache.ibatis.migration.utils.Util;
35  import org.junit.jupiter.api.Test;
36  
37  import uk.org.webcompere.systemstubs.SystemStubs;
38  
39  class NewHookTest {
40  
41    @Test
42    void shouldRunNewHooks() throws Throwable {
43      File basePath = initBaseDir();
44      File scriptPath = new File(basePath.getCanonicalPath() + File.separator + "scripts");
45      String output = SystemStubs.tapSystemOut(() -> {
46        Migrator.main(
47            TestUtil.args("--path=" + basePath.getAbsolutePath(), "--idpattern=00", "new", "create table1 JIRA-123"));
48      });
49      String[] scripts = scriptPath.list();
50      assertEquals(4, scripts.length);
51      assertTrue(output.contains("SUCCESS"));
52      assertTrue(output.contains("Description is valid."));
53      assertTrue(output.contains("Renamed 03_create_table1_JIRA-123.sql to 03_create_table1_JIRA123.sql"));
54      assertTrue(new File(scriptPath, "03_create_table1_JIRA123.sql").exists());
55      assertTrue(TestUtil.deleteDirectory(basePath), "delete test dir");
56    }
57  
58    @Test
59    void shouldNotCreateFileWhenBeforeHookThrowsException() throws Throwable {
60      File basePath = initBaseDir();
61      File scriptPath = new File(basePath.getCanonicalPath() + File.separator + "scripts");
62      String output = SystemStubs.tapSystemOut(() -> {
63        int exitCode = SystemStubs.catchSystemExit(() -> {
64          Migrator.main(TestUtil.args("--path=" + basePath.getAbsolutePath(), "new", "create table1"));
65        });
66        assertEquals(1, exitCode);
67      });
68      assertTrue(output.contains("FAILURE"));
69      assertEquals(3, scriptPath.list().length);
70      assertTrue(TestUtil.deleteDirectory(basePath), "delete test dir");
71    }
72  
73    protected File initBaseDir() throws IOException {
74      File basePath = TestUtil.getTempDir();
75      Migrator.main(TestUtil.args("--path=" + basePath.getAbsolutePath(), "--idpattern=00", "init"));
76      // Copy hook script
77      File hooksDir = new File(basePath, "hooks");
78      hooksDir.mkdir();
79      try (
80          FileInputStream srcStream = new FileInputStream(
81              Resources.getResourceAsFile("org/apache/ibatis/migration/hook/testdir/hooks/NewHook.js"));
82          FileOutputStream destStream = new FileOutputStream(Util.file(hooksDir, "NewHook.js"))) {
83        FileChannel srcChannel = srcStream.getChannel();
84        FileChannel destChannel = destStream.getChannel();
85        srcChannel.transferTo(0, srcChannel.size(), destChannel);
86      }
87      // Add hook settings
88      File envFile = new File(basePath.getCanonicalPath() + File.separator + "environments", "development.properties");
89      try (PrintWriter writer = new PrintWriter(
90          new BufferedWriter(new OutputStreamWriter(new FileOutputStream(envFile, true), Charset.forName("utf-8"))))) {
91        writer.println("hook_before_new=js:NewHook.js:_function=validateDesc");
92        writer.println("hook_after_new=js:NewHook.js:_function=renameFile");
93      }
94      return basePath;
95    }
96  }
