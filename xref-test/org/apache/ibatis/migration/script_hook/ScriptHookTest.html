48    @Test
49    void testDoScript() throws Exception {
50      System.setProperty("DB_NAME", "do_script_hook");
51  
52      String output = SystemStubs.tapSystemOut(() -> {
53        Migrator.main(TestUtil.args("--path=" + dir.getAbsolutePath(), "script", "0", "003"));
54      });
55      assertTrue(output.contains("SUCCESS"));
56  
57      List<String> lines = Arrays.asList(output.split(separator));
58      List<Integer> lineNums = new ArrayList<>();
59  
60      lineNums.add(lineIndexOf(lines, "Before hook: false"));
61  
62      lineNums.add(lineIndexOf(lines, "Before each hook: 1/false"));
63      lineNums.add(lineIndexOf(lines, "CREATE TABLE CHANGES .*"));
64      lineNums.add(lineIndexOf(lines,
65          "INSERT INTO CHANGES \\(ID, APPLIED_AT, DESCRIPTION\\) VALUES \\(1, '.*', 'create changelog'\\);"));
66      lineNums.add(lineIndexOf(lines, "After each hook: 1/false"));
67  
68      lineNums.add(lineIndexOf(lines, "Before each hook: 2/false"));
69      lineNums.add(lineIndexOf(lines, "select 'do second migration' from \\(values\\(0\\)\\);"));
70      lineNums.add(lineIndexOf(lines,
71          "INSERT INTO CHANGES \\(ID, APPLIED_AT, DESCRIPTION\\) VALUES \\(2, '.*', 'second migration'\\);"));
72      lineNums.add(lineIndexOf(lines, "After each hook: 2/false"));
73  
74      lineNums.add(lineIndexOf(lines, "Before each hook: 3/false"));
75      lineNums.add(lineIndexOf(lines, "select 'do third migration' from \\(values\\(0\\)\\);"));
76      lineNums.add(lineIndexOf(lines,
77          "INSERT INTO CHANGES \\(ID, APPLIED_AT, DESCRIPTION\\) VALUES \\(3, '.*', 'third migration'\\);"));
78      lineNums.add(lineIndexOf(lines, "After each hook: 3/false"));
79  
80      lineNums.add(lineIndexOf(lines, "After hook: false"));
81  
82      Set<Integer> treeSet = new TreeSet<>(lineNums);
83      assertEquals(treeSet.toString(), lineNums.toString());
84    }

131     System.setProperty("DB_NAME", "undo_script_hook");
132 
133     String output = SystemStubs.tapSystemOut(() -> {
134       Migrator.main(TestUtil.args("--path=" + dir.getAbsolutePath(), "script", "003", "0"));
135     });

137 
138     List<String> lines = Arrays.asList(output.split(separator));
139     List<Integer> lineNums = new ArrayList<>();
140 
141     lineNums.add(lineIndexOf(lines, "Before hook: true"));
142 
143     lineNums.add(lineIndexOf(lines, "Before each hook: 3/true"));
144     lineNums.add(lineIndexOf(lines, "select 'undo third migration' from \\(values\\(0\\)\\);"));
145     lineNums.add(lineIndexOf(lines, "DELETE FROM CHANGES WHERE ID = 3;"));
146     lineNums.add(lineIndexOf(lines, "After each hook: 3/true"));
147 
148     lineNums.add(lineIndexOf(lines, "Before each hook: 2/true"));
149     lineNums.add(lineIndexOf(lines, "select 'undo second migration' from \\(values\\(0\\)\\);"));
150     lineNums.add(lineIndexOf(lines, "DELETE FROM CHANGES WHERE ID = 2;"));
151     lineNums.add(lineIndexOf(lines, "After each hook: 2/true"));
152 
153     lineNums.add(lineIndexOf(lines, "Before each hook: 1/true"));
154     lineNums.add(lineIndexOf(lines, "DROP TABLE CHANGES;"));
155     assertEquals(-1, lineIndexOf(lines, "DELETE FROM CHANGES WHERE ID = 1;"),
156         "There should be no DELETE for 'create changelog' migration. See gh-201.");
157     lineNums.add(lineIndexOf(lines, "After each hook: 1/true"));
158 
159     lineNums.add(lineIndexOf(lines, "After hook: true"));
160 
161     Set<Integer> treeSet = new TreeSet<>(lineNums);
162     assertEquals(treeSet.toString(), lineNums.toString());
163   }

205   private int lineIndexOf(List<String> lines, String regex) {
206     for (int i = 0; i < lines.size(); i++) {
207       if (lines.get(i).matches(regex)) {
208         return i;
209       }
210     }
211     return -1;
212   }
