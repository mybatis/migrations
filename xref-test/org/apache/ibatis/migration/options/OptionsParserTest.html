23  import static org.apache.ibatis.migration.options.Options.HOOKPATH;
24  import static org.apache.ibatis.migration.options.Options.SCRIPTPATH;
25  import static org.apache.ibatis.migration.options.Options.TEMPLATE;
26  import static org.apache.ibatis.migration.options.Options.TRACE;
27  import static org.apache.ibatis.migration.options.OptionsParser.parse;
28  import static org.assertj.core.api.Assertions.assertThat;
29  import static org.junit.jupiter.api.Assertions.assertTrue;
30  
31  import java.io.File;
32  import java.nio.file.Path;
33  
34  import org.junit.jupiter.api.Test;
35  
36  class OptionsParserTest {
37    @Test
38    void testOptions() {
39      final SelectedOptions options = parse(new String[] { option(FORCE), option(TRACE), option(HELP) });
40  
41      assertTrue(options.isForce());
42      assertTrue(options.isTrace());
43      assertTrue(options.needsHelp());
44    }
45  
46    @Test
47    void testEnvAndTemplate() {
48      final String testValue = "test";
49      final String[] args = { valuedOption(ENV, testValue), valuedOption(TEMPLATE, testValue),
50          valuedOption(Options.IDPATTERN, testValue) };
51      final SelectedOptions options = parse(args);
52  
53      assertThat(options.getEnvironment()).isEqualTo(testValue);
54      assertThat(options.getTemplate()).isEqualTo(testValue);
55      assertThat(options.getIdPattern()).isEqualTo(testValue);
56    }
57  
58    @Test
59    void testFileOptions() {
60      final String testFileName = "test";
61      final File testFile = Path.of(testFileName).toFile();
62      final SelectedOptions expectedOptions = new SelectedOptions();
63  
64      final SelectedPaths paths = expectedOptions.getPaths();
65      paths.setBasePath(testFile);
66      paths.setEnvPath(testFile);
67      paths.setScriptPath(testFile);
68      paths.setDriverPath(testFile);
69      paths.setHookPath(testFile);
70  
71      final String[] args = { valuedOption(Options.PATH, testFile.getAbsolutePath()),
72          valuedOption(ENVPATH, testFile.getAbsolutePath()), valuedOption(SCRIPTPATH, testFile.getAbsolutePath()),
73          valuedOption(DRIVERPATH, testFile.getAbsolutePath()), valuedOption(HOOKPATH, testFile.getAbsolutePath()) };
74  
75      final SelectedOptions pathOptions = parse(args);
76      checkFileOptionSet(pathOptions.getPaths().getBasePath(), testFileName);
77      checkFileOptionSet(pathOptions.getPaths().getEnvPath(), testFileName);
78      checkFileOptionSet(pathOptions.getPaths().getScriptPath(), testFileName);
79      checkFileOptionSet(pathOptions.getPaths().getDriverPath(), testFileName);
80      checkFileOptionSet(pathOptions.getPaths().getHookPath(), testFileName);
81    }
82  
83    private void checkFileOptionSet(File aFile, String expectedFileName) {
84      assertThat(aFile).hasName(expectedFileName);
85    }
86  
87    @Test
88    void onlyPopulatesCommandOnce() {
89      final String command = "command";
90      final String ignoredCommand = "ignoredCommand";
91      final String anotherIgnored = "anotherIgnored";
92  
93      final SelectedOptions options = parse(new String[] { command, ignoredCommand, anotherIgnored });
94  
95      assertThat(options.getCommand()).isEqualTo(command);
96      assertThat(options.getParams()).contains(ignoredCommand);
97      assertThat(options.getParams()).contains(anotherIgnored);
98    }
99  
100   private String valuedOption(Options option, String aValue) {
101     return option(option) + '=' + aValue;
102   }
103 
104   private String option(Options option) {
105     return "--" + option;
106   }
107 }
