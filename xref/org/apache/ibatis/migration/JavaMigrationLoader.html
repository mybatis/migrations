16  package org.apache.ibatis.migration;
17  
18  import java.io.Reader;
19  import java.io.StringReader;
20  import java.lang.reflect.Modifier;
21  import java.util.ArrayList;
22  import java.util.List;
23  import java.util.Set;
24  
25  import org.apache.ibatis.io.ResolverUtil;
26  import org.apache.ibatis.io.ResolverUtil.Test;
27  
28  public class JavaMigrationLoader implements MigrationLoader {
29  
30    private String[] packageNames;
31  
32    private ClassLoader classLoader;
33  
34    public JavaMigrationLoader(String... packageNames) {
35      this(null, packageNames);
36    }
37  
38    public JavaMigrationLoader(ClassLoader classLoader, String... packageNames) {
39      super();
40      this.classLoader = classLoader;
41      this.packageNames = packageNames;
42    }
43  
44    @Override
45    public List<Change> getMigrations() {
46      List<Change> migrations = new ArrayList<Change>();
47      ResolverUtil<MigrationScript> resolver = getResolver(MigrationScript.class);
48      resolver.findImplementations(MigrationScript.class, packageNames);
49      Set<Class<? extends MigrationScript>> classes = resolver.getClasses();
50      for (Class<? extends MigrationScript> clazz : classes) {
51        try {
52          if (!Modifier.isAbstract(clazz.getModifiers())) {
53            MigrationScript script = clazz.newInstance();
54            Change change = parseChangeFromMigrationScript(script);
55            migrations.add(change);
56          }
57        } catch (Exception e) {
58          throw new MigrationException("Could not instanciate MigrationScript: " + clazz.getName(), e);
59        }
60      }
61      return migrations;
62    }
63  
64    private Change parseChangeFromMigrationScript(MigrationScript script) {
65      Change change = new Change();
66      change.setId(script.getId());
67      change.setDescription(script.getDescription());
68      change.setFilename(script.getClass().getName());
69      return change;
70    }
71  
72    @Override
73    public Reader getScriptReader(Change change, boolean undo) {
74      ResolverUtil<MigrationScript> resolver = getResolver(MigrationScript.class);
75      final String className = change.getFilename();
76      for (String pkg : packageNames) {
77        resolver.find(new Test() {
78          @Override
79          public boolean matches(Class<?> type) {
80            return type != null && MigrationScript.class.isAssignableFrom(type) && type.getName().equals(className);
81          }
82        }, pkg);
83      }
84      Reader reader = null;
85      Set<Class<? extends MigrationScript>> classes = resolver.getClasses();
86      for (Class<? extends MigrationScript> clazz : classes) {
87        try {
88          MigrationScript script = clazz.newInstance();
89          reader = new StringReader(undo ? script.getDownScript() : script.getUpScript());
90        } catch (Exception e) {
91          throw new MigrationException("Could not instanciate MigrationScript: " + clazz.getName(), e);
92        }
93        // There should be only one script.
94        break;
95      }
96      return reader;
97    }
98  
99    @Override
100   public Reader getBootstrapReader() {
101     return getSoleScriptReader(BootstrapScript.class);
102   }
103 
104   @Override
105   public Reader getOnAbortReader() {
106     return getSoleScriptReader(OnAbortScript.class);
107   }
108 
109   public <T extends SimpleScript> Reader getSoleScriptReader(Class<T> scriptClass) {
110     ResolverUtil<T> resolver = getResolver(scriptClass);
111     resolver.findImplementations(scriptClass, packageNames);
112     Set<Class<? extends T>> classes = resolver.getClasses();
113     if (classes == null || classes.isEmpty()) {
114       return null;
115     }
116     if (classes.size() > 1) {
117       throw new MigrationException("There can be only one implementation of " + scriptClass.getName());
118     }
119     Class<? extends T> clazz = classes.iterator().next();
120     try {
121       T script = clazz.newInstance();
122       return new StringReader(script.getScript());
123     } catch (Exception e) {
124       throw new MigrationException("Could not instanciate script class: " + clazz.getName(), e);
125     }
126   }
127 
128   private <T> ResolverUtil<T> getResolver(Class<T> type) {
129     ResolverUtil<T> resolver = new ResolverUtil<T>();
130     if (classLoader != null) {
131       resolver.setClassLoader(classLoader);
132     }
133     return resolver;
134   }
135 }
