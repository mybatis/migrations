16  package org.apache.ibatis.migration.operations;
17  
18  import java.io.PrintStream;
19  import java.sql.SQLException;
20  import java.util.Collections;
21  import java.util.List;
22  
23  import org.apache.ibatis.jdbc.ScriptRunner;
24  import org.apache.ibatis.jdbc.SqlRunner;
25  import org.apache.ibatis.migration.Change;
26  import org.apache.ibatis.migration.ConnectionProvider;
27  import org.apache.ibatis.migration.MigrationException;
28  import org.apache.ibatis.migration.MigrationLoader;
29  import org.apache.ibatis.migration.options.DatabaseOperationOption;
30  
31  public final class DownOperation extends DatabaseOperation<DownOperation> {
32    private Integer steps;
33  
34    public DownOperation() {
35      this(null);
36    }
37  
38    public DownOperation(Integer steps) {
39      super();
40      this.steps = steps;
41    }
42  
43    @Override
44    public DownOperation operate(ConnectionProvider connectionProvider, MigrationLoader migrationsLoader, DatabaseOperationOption option, PrintStream printStream) {
45      try {
46        if (option == null) {
47          option = new DatabaseOperationOption();
48        }
49        Change lastChange = getLastAppliedChange(connectionProvider, option);
50        if (lastChange == null) {
51          println(printStream, "Changelog exist, but no migration found.");
52        } else {
53          List<Change> migrations = migrationsLoader.getMigrations();
54          Collections.sort(migrations);
55          Collections.reverse(migrations);
56          int stepCount = 0;
57          for (Change change : migrations) {
58            if (change.getId().equals(lastChange.getId())) {
59              println(printStream, horizontalLine("Undoing: " + change.getFilename(), 80));
60              ScriptRunner runner = getScriptRunner(connectionProvider, option, printStream);
61              try {
62                runner.runScript(migrationsLoader.getScriptReader(change, true));
63              } finally {
64                runner.closeConnection();
65              }
66              if (changelogExists(connectionProvider, option)) {
67                deleteChange(connectionProvider, change, option);
68              } else {
69                println(printStream, "Changelog doesn't exist. No further migrations will be undone (normal for the last migration).");
70                stepCount = steps;
71              }
72              println(printStream);
73              stepCount++;
74              if (steps == null || stepCount >= steps) {
75                break;
76              }
77              lastChange = getLastAppliedChange(connectionProvider, option);
78            }
79          }
80        }
81        return this;
82      } catch (Exception e) {
83        throw new MigrationException("Error undoing last migration.  Cause: " + e, e);
84      }
85    }
86  
87    protected void deleteChange(ConnectionProvider connectionProvider, Change change, DatabaseOperationOption option) {
88      SqlRunner runner = getSqlRunner(connectionProvider);
89      try {
90        runner.delete("delete from " + option.getChangelogTable() + " where id = ?", change.getId());
91      } catch (SQLException e) {
92        throw new MigrationException("Error querying last applied migration.  Cause: " + e, e);
93      } finally {
94        runner.closeConnection();
95      }
96    }
97  }
