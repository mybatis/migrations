16  package org.apache.ibatis.migration;
17  
18  import java.io.File;
19  import java.io.IOException;
20  import java.io.Reader;
21  import java.math.BigDecimal;
22  import java.util.ArrayList;
23  import java.util.Arrays;
24  import java.util.List;
25  import java.util.Properties;
26  
27  import org.apache.ibatis.migration.utils.Util;
28  
29  public class FileMigrationLoader implements MigrationLoader {
30    private final File scriptsDir;
31  
32    private final String charset;
33  
34    private final Properties properties;
35  
36    public FileMigrationLoader(File scriptsDir, String charset, Properties properties) {
37      super();
38      this.scriptsDir = scriptsDir;
39      this.charset = charset;
40      this.properties = properties;
41    }
42  
43    @Override
44    public List<Change> getMigrations() {
45      List<Change> migrations = new ArrayList<Change>();
46      if (scriptsDir.isDirectory()) {
47        String[] filenames = scriptsDir.list();
48        if (filenames == null) {
49          throw new MigrationException(scriptsDir + " does not exist.");
50        }
51        Arrays.sort(filenames);
52        for (String filename : filenames) {
53          if (filename.endsWith(".sql") && !"bootstrap.sql".equals(filename)) {
54            Change change = parseChangeFromFilename(filename);
55            migrations.add(change);
56          }
57        }
58      }
59      return migrations;
60    }
61  
62    private Change parseChangeFromFilename(String filename) {
63      try {
64        Change change = new Change();
65        int lastIndexOfDot = filename.lastIndexOf(".");
66        String[] parts = filename.substring(0, lastIndexOfDot).split("_");
67        change.setId(new BigDecimal(parts[0]));
68        StringBuilder builder = new StringBuilder();
69        for (int i = 1; i < parts.length; i++) {
70          if (i > 1) {
71            builder.append(" ");
72          }
73          builder.append(parts[i]);
74        }
75        change.setDescription(builder.toString());
76        change.setFilename(filename);
77        return change;
78      } catch (Exception e) {
79        throw new MigrationException("Error parsing change from file.  Cause: " + e, e);
80      }
81    }
82  
83    @Override
84    public Reader getScriptReader(Change change, boolean undo) {
85      try {
86        return new MigrationReader(Util.file(scriptsDir, change.getFilename()), charset, undo, properties);
87      } catch (IOException e) {
88        throw new MigrationException("Error reading " + change.getFilename(), e);
89      }
90    }
91  
92    @Override
93    public Reader getBootstrapReader() {
94      try {
95        File bootstrap = Util.file(scriptsDir, "bootstrap.sql");
96        if (bootstrap.exists()) {
97          return new MigrationReader(bootstrap, charset, false, properties);
98        }
99        return null;
100     } catch (IOException e) {
101       throw new MigrationException("Error reading bootstrap.sql", e);
102     }
103   }
104 }
