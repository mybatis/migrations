34      copyResourceTo("org/apache/ibatis/migration/template_migration.sql",
35          Util.file(scriptPath, getNextIDAsString() + "_first_migration.sql"),
36          new Properties() {

39            }
40          });

45    protected void ensureDirectoryIsEmpty(File path) {
46      String[] list = path.list();
47      if (list.length != 0) {
48        for (String entry : list) {
49          if (!entry.startsWith(".")) {
50            throw new MigrationException(
51                "Directory must be empty (.svn etc allowed): " + path.getAbsolutePath());
52          }
53        }
54      }
55    }
56  
57    protected void createDirectoryIfNecessary(File path) {
58      if (!path.exists()) {
59        printStream.println("Creating: " + path.getName());
60        if (!path.mkdirs()) {
61          throw new MigrationException(
62              "Could not create directory path for an unknown reason. Make sure you have access to the directory.");
63        }
64      }
65    }
