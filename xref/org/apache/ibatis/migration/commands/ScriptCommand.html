24  import java.util.List;
25  import java.util.Map;

28  import org.apache.ibatis.migration.Change;
29  import org.apache.ibatis.migration.MigrationException;
30  import org.apache.ibatis.migration.hook.MigrationHook;

36  public final class ScriptCommand extends BaseCommand {
37  
38    public ScriptCommand(SelectedOptions options) {
39      super(options);
40    }
41  
42    @Override
43    public void execute(String... sparams) {
44      try {
45        if (sparams == null || sparams.length < 1 || sparams[0] == null) {
46          throw new MigrationException("The script command requires a range of versions from v1 - v2.");
47        }
48        StringTokenizer parser = new StringTokenizer(sparams[0]);
49        int tokenCount = parser.countTokens();
50        boolean scriptPending = false;
51        boolean scriptPendingUndo = false;
52  
53        String firstToken = parser.nextToken();
54  
55        if (tokenCount == 1 && firstToken.equals("pending")) {
56          scriptPending = true;
57        } else if (tokenCount == 1 && firstToken.equals("pending_undo")) {
58          scriptPendingUndo = true;
59        } else if (!scriptPending && !scriptPendingUndo && tokenCount != 2) {
60          throw new MigrationException("The script command requires a range of versions from v1 - v2.");
61        }
62  
63        BigDecimal v1 = (scriptPending || scriptPendingUndo) ? null : new BigDecimal(firstToken);
64        BigDecimal v2 = (scriptPending || scriptPendingUndo) ? null : new BigDecimal(parser.nextToken());
65  
66        boolean undo;
67        undo = scriptPendingUndo;
68        if (!scriptPending && !scriptPendingUndo) {
69          int comparison = v1.compareTo(v2);
70          if (comparison == 0) {
71            throw new MigrationException(
72                "The script command requires two different versions. Use 0 to include the first version.");
73          }
74          undo = comparison > 0;
75        }
76  
77        Map<String, Object> hookBindings = new HashMap<>();
78        MigrationHook hook = createScriptHook();
79        List<Change> migrations = (scriptPending || scriptPendingUndo) ? new StatusOperation()
80            .operate(getConnectionProvider(), getMigrationLoader(), getDatabaseOperationOption(), null).getCurrentStatus()
81            : getMigrationLoader().getMigrations();
82        Collections.sort(migrations);
83        if (undo) {
84          Collections.reverse(migrations);
85        }
86        int count = 0;
87        for (int i = 0; i < migrations.size(); i++) {
88          Change change = migrations.get(i);
89          if (shouldRun(change, v1, v2, scriptPending || scriptPendingUndo)) {
90            if (count == 0 && hook != null) {
91              hookBindings.put(MigrationHook.HOOK_CONTEXT, new ScriptHookContext(null, undo));
92              hook.before(hookBindings);
93              printStream.println();
94            }
95            if (hook != null) {
96              hookBindings.put(MigrationHook.HOOK_CONTEXT, new ScriptHookContext(change.clone(), undo));
97              hook.beforeEach(hookBindings);
98              printStream.println();
99            }
100           printStream.println("-- " + change.getFilename());
101           Reader migrationReader = getMigrationLoader().getScriptReader(change, undo);
102           char[] cbuf = new char[1024];
103           int l;
104           while ((l = migrationReader.read(cbuf)) > -1) {
105             printStream.print(l == cbuf.length ? cbuf : Arrays.copyOf(cbuf, l));
106           }
107           count++;
108           printStream.println();
109           printStream.println();
110           if (!undo) {
111             printStream.println(generateVersionInsert(change));
112           } else if (i + 1 < migrations.size() || !DESC_CREATE_CHANGELOG.equals(change.getDescription())) {
113             printStream.println(generateVersionDelete(change));
114           }
115           printStream.println();
116           if (hook != null) {
117             hookBindings.put(MigrationHook.HOOK_CONTEXT, new ScriptHookContext(change.clone(), undo));
118             hook.afterEach(hookBindings);
119             printStream.println();
120           }
121         }
122       }
123       if (count > 0 && hook != null) {
124         hookBindings.put(MigrationHook.HOOK_CONTEXT, new ScriptHookContext(null, undo));
125         hook.after(hookBindings);
126         printStream.println();
127       }
128     } catch (IOException e) {
129       throw new MigrationException("Error generating script. Cause: " + e, e);
130     }
131   }
132 
133   private String generateVersionInsert(Change change) {
134     return "INSERT INTO " + changelogTable() + " (ID, APPLIED_AT, DESCRIPTION) " + "VALUES (" + change.getId() + ", '"
135         + DatabaseOperation.generateAppliedTimeStampAsString() + "', '" + change.getDescription().replace('\'', ' ')
136         + "')" + getDelimiter();
137   }
138 
139   private String generateVersionDelete(Change change) {
140     return "DELETE FROM " + changelogTable() + " WHERE ID = " + change.getId() + getDelimiter();
141   }
142 
143   private boolean shouldRun(Change change, BigDecimal v1, BigDecimal v2, boolean pendingOnly) {
144     if (!pendingOnly) {
145       BigDecimal id = change.getId();
146       if (v1.compareTo(v2) > 0) {
147         return (id.compareTo(v2) > 0 && id.compareTo(v1) <= 0);
148       } else {
149         return (id.compareTo(v1) > 0 && id.compareTo(v2) <= 0);
150       }
151     } else {
152       return change.getAppliedTimestamp() == null;
153     }
154   }
155 
156   // Issue 699
157   private String getDelimiter() {
158     StringBuilder delimiter = new StringBuilder();
159     if (environment().isFullLineDelimiter()) {
160       delimiter.append("\n");
161     }
162     delimiter.append(environment().getDelimiter());
163     return delimiter.toString();
164   }
165 
166   private MigrationHook createScriptHook() {
167     String before = environment().getHookBeforeScript();
168     String beforeEach = environment().getHookBeforeEachScript();
169     String afterEach = environment().getHookAfterEachScript();
170     String after = environment().getHookAfterScript();
171     if (before == null && beforeEach == null && afterEach == null && after == null) {
172       return null;
173     }
174     return createFileMigrationHook(before, beforeEach, afterEach, after);
175   }
