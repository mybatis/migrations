16  package org.apache.ibatis.migration.commands;
17  
18  import java.io.IOException;
19  import java.io.InputStream;
20  import java.io.PrintStream;
21  import java.nio.charset.Charset;
22  import java.util.Locale;
23  import java.util.Properties;
24  
25  public final class InfoCommand implements Command {
26    private final PrintStream out;
27  
28    public InfoCommand(PrintStream out) {
29      this.out = out;
30    }
31  
32    @Override
33    public void execute(String... params) {
34      Properties properties = new Properties();
35      try (InputStream is = getClass().getResourceAsStream("/mybatis-migrations.properties")) {
36        if (is != null) {
37          properties.load(is);
38        }
39      } catch (IOException e) {
40        // ignore
41      }
42  
43      out.printf("%s %s (%s)%n", properties.getProperty("name"), properties.getProperty("version"),
44          properties.getProperty("build"));
45      out.printf("Java version: %s, vendor: %s%n", System.getProperty("java.version"), System.getProperty("java.vendor"));
46      out.printf("Java home: %s%n", System.getProperty("java.home"));
47      out.printf("Default locale: %s, platform encoding: %s%n", Locale.getDefault().toLanguageTag(),
48          Charset.defaultCharset().name());
49      out.printf("OS name: \"%s\", version: \"%s\", arch: \"%s\", family: \"%s\"%n", System.getProperty("os.name"),
50          System.getProperty("os.version"), System.getProperty("os.arch"), getOsFamily());
51    }
52  
53    private static final String getOsFamily() {
54      String osName = System.getProperty("os.name").toLowerCase();
55      String pathSep = System.getProperty("path.separator");
56  
57      if (osName.indexOf("windows") != -1) {
58        return "windows";
59      } else if (osName.indexOf("os/2") != -1) {
60        return "os/2";
61      } else if (osName.indexOf("z/os") != -1 || osName.indexOf("os/390") != -1) {
62        return "z/os";
63      } else if (osName.indexOf("os/400") != -1) {
64        return "os/400";
65      } else if (pathSep.equals(";")) {
66        return "dos";
67      } else if (osName.indexOf("mac") != -1) {
68        if (osName.endsWith("x")) {
69          return "mac"; // MACOSX
70        }
71        return "unix";
72      } else if (osName.indexOf("nonstop_kernel") != -1) {
73        return "tandem";
74      } else if (osName.indexOf("openvms") != -1) {
75        return "openvms";
76      } else if (pathSep.equals(":")) {
77        return "unix";
78      }
79  
80      return "undefined";
81    }
82  
83  }
