16  package org.apache.ibatis.migration.commands;
17  
18  import java.io.IOException;
19  import java.io.InputStream;
20  import java.io.PrintStream;
21  import java.util.Properties;
22  
23  public final class InfoCommand implements Command {
24    private final PrintStream out;
25  
26    public InfoCommand(PrintStream out) {
27      this.out = out;
28    }
29  
30    @Override
31    public void execute(String... params) {
32      Properties properties = new Properties();
33      InputStream input = getClass().getClassLoader()
34          .getResourceAsStream("META-INF/maven/org.mybatis/mybatis-migrations/pom.properties");
35  
36      if (input != null) {
37        try {
38          properties.load(input);
39        } catch (IOException e) {
40          // ignore, just don't load the properties
41        } finally {
42          try {
43            input.close();
44          } catch (IOException e) {
45            // close quietly
46          }
47        }
48      }
49  
50      out.printf("%s %s (%s)%n", properties.getProperty("name"), properties.getProperty("version"),
51          properties.getProperty("build"));
52      out.printf("Java version: %s, vendor: %s%n", System.getProperty("java.version"), System.getProperty("java.vendor"));
53      out.printf("Java home: %s%n", System.getProperty("java.home"));
54      out.printf("Default locale: %s_%s, platform encoding: %s%n", System.getProperty("user.language"),
55          System.getProperty("user.country"), System.getProperty("sun.jnu.encoding"));
56      out.printf("OS name: \"%s\", version: \"%s\", arch: \"%s\", family: \"%s\"%n", System.getProperty("os.name"),
57          System.getProperty("os.version"), System.getProperty("os.arch"), getOsFamily());
58    }
59  
60    private static final String getOsFamily() {
61      String osName = System.getProperty("os.name").toLowerCase();
62      String pathSep = System.getProperty("path.separator");
63  
64      if (osName.indexOf("windows") != -1) {
65        return "windows";
66      } else if (osName.indexOf("os/2") != -1) {
67        return "os/2";
68      } else if (osName.indexOf("z/os") != -1 || osName.indexOf("os/390") != -1) {
69        return "z/os";
70      } else if (osName.indexOf("os/400") != -1) {
71        return "os/400";
72      } else if (pathSep.equals(";")) {
73        return "dos";
74      } else if (osName.indexOf("mac") != -1) {
75        if (osName.endsWith("x")) {
76          return "mac"; // MACOSX
77        }
78        return "unix";
79      } else if (osName.indexOf("nonstop_kernel") != -1) {
80        return "tandem";
81      } else if (osName.indexOf("openvms") != -1) {
82        return "openvms";
83      } else if (pathSep.equals(":")) {
84        return "unix";
85      }
86  
87      return "undefined";
88    }
89  
90  }
