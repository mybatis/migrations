16  package org.apache.ibatis.migration;
17  
18  import static org.apache.ibatis.migration.options.OptionsParser.parse;
19  
20  import java.io.File;
21  import java.io.PrintStream;
22  import java.util.Date;
23  
24  import org.apache.ibatis.migration.commands.Command;
25  import org.apache.ibatis.migration.commands.Commands;
26  import org.apache.ibatis.migration.commands.InfoCommand;
27  import org.apache.ibatis.migration.commands.InitializeCommand;
28  import org.apache.ibatis.migration.options.Options;
29  import org.apache.ibatis.migration.options.SelectedOptions;
30  import org.apache.ibatis.migration.utils.Util;
31  
32  public class CommandLine {
33    private final PrintStream console = System.out;
34    private final String[] args;
35  
36    public CommandLine(String[] args) {
37      this.args = args;
38    }
39  
40    public void execute() {
41      final SelectedOptions selectedOptions = parse(args);
42      if (selectedOptions.needsHelp()) {
43        printUsage();
44        return;
45      } else if (selectedOptions.getCommand() == null) {
46        console.printf("No command specified.%n");
47        printUsage();
48        return;
49      }
50      try {
51        Command command = Commands.resolveCommand(selectedOptions.getCommand(), selectedOptions);
52        if (command instanceof InitializeCommand || command instanceof InfoCommand
53            || validBasePath(selectedOptions.getPaths().getBasePath())) {
54          runCommand(command, selectedOptions);
55        }
56      } catch (Exception e) {
57        String errorMessage = e.getMessage();
58  
59        if (hasColor(selectedOptions)) {
60          console.printf(ConsoleColors.RED + "\nERROR: %s%n", errorMessage + ConsoleColors.RESET);
61        } else {
62          console.printf("\nERROR: %s%n", errorMessage);
63        }
64  
65        if (selectedOptions.isTrace()) {
66          e.printStackTrace();
67        }
68        System.exit(1); // Issue 730
69      }
70    }
71  
72    private void runCommand(Command command, SelectedOptions selectedOptions) {
73      console.printf("------------------------------------------------------------------------%n");
74      console.printf("-- MyBatis Migrations - %s%n", selectedOptions.getCommand());
75      console.printf("------------------------------------------------------------------------%n");
76  
77      long start = System.currentTimeMillis();
78      boolean exceptionCaught = false;
79  
80      try {
81        command.execute(selectedOptions.getParams());
82      } catch (Throwable t) {
83        exceptionCaught = true;
84        if (t instanceof MigrationException) {
85          throw (MigrationException) t;
86        } else {
87          throw new MigrationException(t);
88        }
89      } finally {
90        console.printf("------------------------------------------------------------------------%n");
91  
92        if (hasColor(selectedOptions)) {
93          console.printf("-- MyBatis Migrations %s%s%s%n", (exceptionCaught) ? ConsoleColors.RED : ConsoleColors.GREEN,
94              (exceptionCaught) ? "FAILURE" : "SUCCESS", ConsoleColors.RESET);
95        } else {
96          console.printf("-- MyBatis Migrations %s%n", (exceptionCaught) ? "FAILURE" : "SUCCESS");
97        }
98  
99        console.printf("-- Total time: %ss%n", ((System.currentTimeMillis() - start) / 1000));
100       console.printf("-- Finished at: %s%n", new Date());
101       printMemoryUsage();
102       console.printf("------------------------------------------------------------------------%n");
103     }
104   }
105 
106   protected boolean hasColor(SelectedOptions selectedOptions) {
107     return selectedOptions.hasColor() || Util.getPropertyOptionAsBoolean(Options.COLOR.toString().toLowerCase());
108   }
109 
110   private void printMemoryUsage() {
111     final Runtime runtime = Runtime.getRuntime();
112     final int megaUnit = 1024 * 1024;
113     final long usedMemory = (runtime.totalMemory() - runtime.freeMemory()) / megaUnit;
114     final long totalMemory = runtime.totalMemory() / megaUnit;
115 
116     console.printf("-- Final Memory: %sM/%sM%n", usedMemory, totalMemory);
117   }
118 
119   private boolean validBasePath(File basePath) {
120     final boolean validDirectory = basePath.exists() && basePath.isDirectory();
121 
122     if (!validDirectory) {
123       console.printf("Migrations path must be a directory: %s%n", basePath.getAbsolutePath());
124     }
125 
126     return validDirectory;
127   }
128 
129   private void printUsage() {
130     console.printf(
131         "%nUsage: migrate command [parameter] [--path=<directory>] [--env=<environment>] [--template=<path to custom template>]%n%n");
132     console.printf("--path=<directory>   Path to repository.  Default current working directory.%n");
133     console.printf("--env=<environment>  Environment to configure. Default environment is 'development'.%n");
134     console.printf("--template=<template>  Path to custom template for creating new sql scripts.%n");
135     console.printf("--force              Forces script to continue even if SQL errors are encountered.%n");
136     console.printf("--help               Displays this usage message.%n");
137     console.printf("--trace              Shows additional error details (if any).%n");
138     console.printf("--quiet              Suppresses output.%n");
139     console.printf("--color              Colorize output.%n");
140     console.printf("%n");
141     console.printf("Commands:%n");
142     console.printf("  info               Display build version informations.%n");
143     console.printf("  init               Creates (if necessary) and initializes a migration path.%n");
144     console.printf("  bootstrap          Runs the bootstrap SQL script (see scripts/bootstrap.sql for more).%n");
145     console.printf("  new <description>  Creates a new migration with the provided description.%n");
146     console.printf("  up [n]             Run unapplied migrations, ALL by default, or 'n' specified.%n");
147     console
148         .printf("  down [n]           Undoes migrations applied to the database. ONE by default or 'n' specified.%n");
149     console.printf("  version <version>  Migrates the database up or down to the specified version.%n");
150     console.printf("  pending            Force executes pending migrations out of order (not recommended).%n");
151     console.printf("  status             Prints the changelog from the database if the changelog table exists.%n");
152     console
153         .printf("  script <v1> <v2>   Generates a delta migration script from version v1 to v2 (undo if v1 > v2).%n");
154     console.printf("%n");
155     console.printf("  * Shortcuts are accepted by using the first few (unambiguous) letters of each command..%n");
156     console.printf("%n");
157   }
158 }
