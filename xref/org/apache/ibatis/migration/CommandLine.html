16  package org.apache.ibatis.migration;
17  
18  import static org.apache.ibatis.migration.commands.Commands.resolveCommand;
19  import static org.apache.ibatis.migration.options.OptionsParser.parse;
20  
21  import java.io.File;
22  import java.io.PrintStream;
23  import java.util.Date;
24  
25  import org.apache.ibatis.migration.commands.Command;
26  import org.apache.ibatis.migration.options.Options;
27  import org.apache.ibatis.migration.options.SelectedOptions;
28  import org.apache.ibatis.migration.utils.Util;
29  import org.apache.ibatis.migration.ConsoleColors;
30  
31  public class CommandLine {
32    private final PrintStream console = System.out;
33    private final String[] args;
34  
35    public CommandLine(String[] args) {
36      this.args = args;
37    }
38  
39    public void execute() {
40      final SelectedOptions selectedOptions = parse(args);
41      try {
42        if (!validOptions(selectedOptions) || selectedOptions.needsHelp()) {
43          printUsage();
44        } else {
45          runCommand(selectedOptions);
46        }
47      } catch (Exception e) {
48        String errorMessage = e.getMessage();
49  
50        if (hasColor(selectedOptions)) {
51          console.printf(ConsoleColors.RED + "\nERROR: %s%n", errorMessage + ConsoleColors.RESET);
52        } else {
53          console.printf("\nERROR: %s%n", errorMessage);
54        }
55  
56        if (selectedOptions.isTrace()) {
57          e.printStackTrace();
58        }
59        System.exit(1); // Issue 730
60      }
61    }
62  
63    private void runCommand(SelectedOptions selectedOptions) {
64      final String commandString = selectedOptions.getCommand();
65  
66      console.printf("------------------------------------------------------------------------%n");
67      console.printf("-- MyBatis Migrations - %s%n", commandString);
68      console.printf("------------------------------------------------------------------------%n");
69  
70      long start = System.currentTimeMillis();
71      boolean exceptionCaught = false;
72  
73      try {
74        final Command command = resolveCommand(commandString.toUpperCase(), selectedOptions);
75        command.execute(selectedOptions.getParams());
76      } catch (Throwable t) {
77        exceptionCaught = true;
78        if (t instanceof MigrationException) {
79          throw (MigrationException) t;
80        } else {
81          throw new MigrationException(t);
82        }
83      } finally {
84        console.printf("------------------------------------------------------------------------%n");
85  
86        if (hasColor(selectedOptions)) {
87          console.printf("-- MyBatis Migrations %s%s%s%n", (exceptionCaught) ? ConsoleColors.RED : ConsoleColors.GREEN,
88              (exceptionCaught) ? "FAILURE" : "SUCCESS", ConsoleColors.RESET);
89        } else {
90          console.printf("-- MyBatis Migrations %s%n", (exceptionCaught) ? "FAILURE" : "SUCCESS");
91        }
92  
93        console.printf("-- Total time: %ss%n", ((System.currentTimeMillis() - start) / 1000));
94        console.printf("-- Finished at: %s%n", new Date());
95        printMemoryUsage();
96        console.printf("------------------------------------------------------------------------%n");
97      }
98    }
99  
100   protected boolean hasColor(SelectedOptions selectedOptions) {
101     return selectedOptions.hasColor() || Util.getPropertyOptionAsBoolean(Options.COLOR.toString().toLowerCase());
102   }
103 
104   private void printMemoryUsage() {
105     final Runtime runtime = Runtime.getRuntime();
106     final int megaUnit = 1024 * 1024;
107     final long usedMemory = (runtime.totalMemory() - runtime.freeMemory()) / megaUnit;
108     final long totalMemory = runtime.totalMemory() / megaUnit;
109 
110     console.printf("-- Final Memory: %sM/%sM%n", usedMemory, totalMemory);
111   }
112 
113   private boolean validOptions(SelectedOptions selectedOptions) {
114     if (!selectedOptions.needsHelp() && selectedOptions.getCommand() == null) {
115       console.printf("No command specified.%n");
116       return false;
117     }
118 
119     return validBasePath(selectedOptions.getPaths().getBasePath());
120   }
121 
122   private boolean validBasePath(File basePath) {
123     final boolean validDirectory = basePath.exists() && basePath.isDirectory();
124 
125     if (!validDirectory) {
126       console.printf("Migrations path must be a directory: %s%n", basePath.getAbsolutePath());
127     }
128 
129     return validDirectory;
130   }
131 
132   private void printUsage() {
133     console.printf(
134         "%nUsage: migrate command [parameter] [--path=<directory>] [--env=<environment>] [--template=<path to custom template>]%n%n");
135     console.printf("--path=<directory>   Path to repository.  Default current working directory.%n");
136     console.printf("--env=<environment>  Environment to configure. Default environment is 'development'.%n");
137     console.printf("--template=<template>  Path to custom template for creating new sql scripts.%n");
138     console.printf("--force              Forces script to continue even if SQL errors are encountered.%n");
139     console.printf("--help               Displays this usage message.%n");
140     console.printf("--trace              Shows additional error details (if any).%n");
141     console.printf("--quiet              Suppresses output.%n");
142     console.printf("--color              Colorize output.%n");
143     console.printf("%n");
144     console.printf("Commands:%n");
145     console.printf("  info               Display build version informations.%n");
146     console.printf("  init               Creates (if necessary) and initializes a migration path.%n");
147     console.printf("  bootstrap          Runs the bootstrap SQL script (see scripts/bootstrap.sql for more).%n");
148     console.printf("  new <description>  Creates a new migration with the provided description.%n");
149     console.printf("  up [n]             Run unapplied migrations, ALL by default, or 'n' specified.%n");
150     console
151         .printf("  down [n]           Undoes migrations applied to the database. ONE by default or 'n' specified.%n");
152     console.printf("  version <version>  Migrates the database up or down to the specified version.%n");
153     console.printf("  pending            Force executes pending migrations out of order (not recommended).%n");
154     console.printf("  status             Prints the changelog from the database if the changelog table exists.%n");
155     console
156         .printf("  script <v1> <v2>   Generates a delta migration script from version v1 to v2 (undo if v1 > v2).%n");
157     console.printf("%n");
158     console.printf("  * Shortcuts are accepted by using the first few (unambiguous) letters of each command..%n");
159     console.printf("%n");
160   }
161 }
